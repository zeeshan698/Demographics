pipeline {
    agent any

    stages {
        
        stage('SCM') {
            steps{
               git url: 'https://github.com/vicky-devops/Demographics.git'

            }
        }
        stage('Installing Dependencies') {
            steps {
                    withEnv(["HOME=${env.WORKSPACE}"]) {
                       sh 'python3.8 -m pip install --user --upgrade pip'
                       
                       sh 'python3.8 -m pip install --user -r requirements.txt'
                    } 
            }
        }
        
        stage('ML Modal train deploy') {
            steps {
              
                sh 'python3.8 ./drug_demographic_2-Copy1.py'
            }
        }
    }
    
}
